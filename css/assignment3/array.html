import java.util.Scanner; public class MatrixSwapDiagonal { public static void main(String[] args) { Scanner sc = new Scanner(System.in); int n = sc.nextInt(); int m = sc.nextInt(); int[][] matrix = new int[n][m]; for (int i = 0; i < n; i++) { for (int j = 0; j < m; j++) { matrix[i][j] = sc.nextInt(); } } for (int i = 0; i < n - 1; i++) { for (int j = 0; j < m - 1; j++) { int temp = matrix[i][j]; matrix[i][j] = matrix[i + 1][j + 1]; matrix[i + 1][j + 1] = temp; } } for (int i = 0; i < n; i++) { for (int j = 0; j < m; j++) { System.out.print(matrix[i][j] + " "); } System.out.println(); } sc.close(); } }
